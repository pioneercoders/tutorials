  # Function to print chessboard pattern
def print_chessboard(size=8):
    for i in range(size):
        row = ""
        for j in range(size):
            # Alternate between black (B) and white (W)
            if (i + j) % 2 == 0:
                row += "B "  # Black square
            else:
                row += "W "  # White square
        print(row)

# Call the function to print the chessboard
print_chessboard()
